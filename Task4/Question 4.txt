namespace task4
{
    internal class Program
    {
        static void Main(string[] args)
        {
            User user = new User();
            UserSnapshot snapshot = new UserSnapshot();
            user.display();
            snapshot.display();
            Program p = new Program();
            p.modify_class(user);
            user.display();
            //p.modify_struct(snapshot);
            //snapshot.display();
            p.modify_struct(ref snapshot);
            snapshot.display();
        }

        public void modify_class(User u)
        {
            u.Name = "Ahmed";
        }

        public void modify_struct(ref UserSnapshot u)
        {
            u.Name = "Ahmed";
        }
    }

    class User
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public User()
        {
            Id = 1;
            Name = "Ali";
        }
        public void display()
        {
            Console.WriteLine(Id);
            Console.WriteLine(Name);
        }
    }
    struct UserSnapshot
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public UserSnapshot()
        {
            Id = 2;
            Name = "Mohamed";
        }
        public void display()
        {
            Console.WriteLine(Id);
            Console.WriteLine(Name);
        }
    }
}

->Explaination: 

Class:
refrence data type ,when we pass it to a method and modify values it affects the original value.
its values stored at heap and address stored at stack that points to the values at heap.

Struct:
value data type ,when we pass it to a method and modify values it doesn't affect the original value.
if we want change values we can pass struct by refrence using 'ref' key word.
stored at stack.
