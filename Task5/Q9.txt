namespace Linq_Task
{
    internal class Program
    {
        static void Main(string[] args)
        {
            List<int> nums = [1, 2, 3, 4, 5];
            var query = nums.Where(n => n > 2).ToList(); // ← query defined here
            nums.Add(10); // ← source modified AFTER query
            foreach (var n in query)
                Console.Write(n + " ");

            // -OutPut: 3,4,5,10  --> because where use deferred excution (delay execution and executed at runtime).
            // -If we use .ToList() code will work in immediate excution (compile time) and values will store in 'query'
            //   before adding 10 and result will be 3,4,5.
            // Aggregate / All / Any
        }


    }
}
